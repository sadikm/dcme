    <button id="btn">Click Me</button>
    <div id="output"></div>
    
    <script>
        let btn = document.getElementById("btn");
        let output = document.getElementById("output");
        
        btn.addEventListener("click", function() {
            output.innerHTML = "<p>Button clicked using addEventListener</p>";
        });
    </script>
